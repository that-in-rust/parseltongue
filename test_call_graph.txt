fn main() {
    let user = create_user("Alice".to_string());
    user.display();
    process_data(&user);
}

fn create_user(name: String) -> User {
    User { name }
}

fn process_data(user: &User) {
    println!("Processing: {}", user.get_name());
}

pub struct User {
    name: String,
}

impl User {
    pub fn display(&self) {
        println!("User: {}", self.name);
    }

    pub fn get_name(&self) -> &str {
        &self.name
    }
}
